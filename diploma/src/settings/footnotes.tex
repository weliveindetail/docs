
% Fußnoten in gleitenden Umgebungen
%\usepackage{footnote}

% Das Paket footnote stellt den folgenden Befehl zur Verfügung, um aus 
% bestehenden Umgebungen (figure) eine neue (figure*) zu erzeugen, die
% die Fußnotensteuerung aus dem footnote-Paket beherrscht:
%\makesavenoteenv[figure*]{figure}
%\makesavenoteenv[lstlisting*]{lstlisting}

% Beschriftungen beliebiger Objekte (in Schriftgröße footnotesize)
\usepackage[font=footnotesize]{caption}
